% bigimage{file}
\renewcommand{\bigimage}[2][1.0]{%
    {%
        \setbeamertemplate{background canvas}{ %
            \setbeamercolor{background canvas}{bg=black}%
        }%
        {%
            \begin{frame}%
                {%
                \makebox[\textwidth][c]{%
                  \includegraphics[height=#1\paperheight, width=#1\paperwidth,%
                                   keepaspectratio]{#2}%
                  }%
                }%
            \end{frame}%
        }%
    }%
}

% Timed image transition
\newcommand{\transition}[3][1.0]{%
    {%
        \setbeamertemplate{background canvas}{ %
            \tikz[remember picture, overlay]\node at (current page.center) {
                 \includegraphics[height=#1\paperheight, width=#1\paperwidth, keepaspectratio]
                 {#3}%
            };
        }%
        {%
            \begin{frame}%
                \transdissolve %
                \transduration{#2} %
            \end{frame}%
        }%
    }%
    \begin{frame}%
        \transdissolve %
        \transduration{0.2} %
    \end{frame}%
}
