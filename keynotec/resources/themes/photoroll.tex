\renewcommand{\coverframe}{%
    {%
        \usebackgroundtemplate{%
            \includegraphics[width=\paperwidth]{images/film_strip.png}%
        }%
        \setbeamercolor{title}{fg=white}
        \setbeamercolor{subtitle}{fg=white}
        \setbeamercolor{author}{fg=white}
        \setbeamercolor{institute}{fg=white}
        \setbeamercolor{date}{fg=white}
        \begin{frame}%
            \titlepage%
        \end{frame}%
    }%
}

\renewcommand{\finalframe}[2][Thank you!]{%
    {%
        \usebackgroundtemplate{%
             \includegraphics[width=\paperwidth]{images/film_strip.png}%
        }%
        \begin{frame}%
            \begin{flushright}%
                \color{white}%
                    \huge \textbf{#1}%
                    \vfill%
                    \large \textbf{#2}%
            \end{flushright}%
        \end{frame}%
    }%
}

}

\newcommand{\bighorizontal}[1]{%

        }%

    }%
}

\newcommand{\bigvertical}[1]{%
    {%
        \usebackgroundtemplate{
            \includegraphics[height=\paperheight]{#1}%
        }%
        \begin{frame}%
        \end{frame}%
    }%
}

\newcommand{\imagehorizontal}[2]{%
    {%
        \usebackgroundtemplate{
            \vbox to \paperheight{%
                \vfil\hbox to \paperwidth{%
                    \hfil\includegraphics[width=\paperwidth]{#2}\hfil %
                }%
                \vfil%
            }%
        }%
        \begin{frame}%
            \frametitle{#1}%
        \end{frame}%
    }%
}

\newcommand{\imagevertical}[2]{%
    {%
        \usebackgroundtemplate{
            \vspace{2cm} %
            \vbox to \paperheight{%
                \vfill\vfill\hbox to \paperwidth{%
                    \hfil\includegraphics[height=0.86\paperheight]{#2}\hfil %
                }%
            }%
        }%
        \begin{frame}%
            \frametitle{#1}%
        \end{frame}%
    }%
}
