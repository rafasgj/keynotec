\newcommand{\rightpagenumber}[0]{
    \vspace{6pt}%
    \hspace*{\fill}%
    \usebeamercolor[fg]{page number in head/foot}%
    \usebeamerfont{page number in head/foot}%
    \insertpagenumber%
    \hspace{4pt}%
    \vspace{6pt}%
}

        \begin{flushright}%

% bigtitle{title}
\newcommand{\bigtitle}[1]{%
    \begin{frame}%
        \begin{center}%
            \Huge {#1}%
        \end{center}%
    \end{frame}%
}

% citation{cite}{author}
\renewcommand{\citation}[2]{%
    \begin{frame}%
        \begin{center}%
            \vspace{1cm}
            \large \textit{"#1"}\\%
            \vspace{1cm}
            \footnotesize {#2}%
        \end{center}%
    \end{frame}%
}

% bigimage{file}
\newcommand{\bigimage}[2][1.0]{%
    {%
        %\usebackgroundtemplate{}%
        \begin{frame}%
            {%
            \makebox[\textwidth][c]{%
              \includegraphics[height=#1\paperheight, width=#1\paperwidth,%
                               keepaspectratio]{#2}%
              }%
            }%
        \end{frame}%
    }%
}

% twoimages{file}{file}
\newcommand{\twoimages}[2]{%
    {%
        \begin{frame}%
            \begin{columns}%
                \begin{column}{0.45\paperwidth}%
                    {%
                    \makebox[0.45\paperwidth][c]{%
                      \includegraphics[height=0.99\paperheight, width=0.45\paperwidth,%
                                       keepaspectratio]{#1}%
                      }%
                    }%
                \end{column}%
                \begin{column}{0.45\paperwidth}%
                    {%
                    \makebox[0.45\paperwidth][c]{%
                      \includegraphics[height=0.99\paperheight, width=0.45\paperwidth,%
                                       keepaspectratio]{#2}%
                      }%
                    }%
                \end{column}%
            \end{columns}%
        \end{frame}%
    }%
}
